## 守护进程
```
from threading import Thread
import time
import os
def task(name):
    print('%s is running ' % name)
    time.sleep(1)
    print('%s is over ' % name)

def test():
    time.sleep(3)
    print('over')

if __name__ == '__main__':
    t = Thread(target=task,args=('lichang',))
    test = Thread(target=test)

    t.daemon = True # 开启守护线程
    t.start()
    test.start()
    print('主')
```
## 线程互斥锁
```
from threading import Thread, Lock
import time
money = 100
mutex = Lock()
def task():
    global money
    mutex.acquire()
    tmp = money
    time.sleep(0.1)
    money = tmp - 1
    mutex.release()

if __name__ == '__main__':
    # mutex = Lock()

    t_list = []

    for i in range(100):
        t = Thread(target=task)
        t.start()
        t_list.append(t)

    for t in t_list:
        t.join()
    print(money)

'''数据错乱，加锁处理'''
```
